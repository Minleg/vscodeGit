About

This program allows the user to choose from a set of three commands view, add and exit.

View allows the user to view the content of a file containing username and email address information

Add allows you to add username and email information. Exit allows you to exit the program

                    import os, pyinputplus as pyp
                    def main():
                    try:
                        if os.path.isfile('directory.txt') is False:# checks if the path to the file exists or not
                        myFile = open('directory.txt','w') # 'w' for write mode
                        myFile.write('mmouse mickeymouse@gmail.com\n') # first entry to the file
                        myFile.close() # closes the file
                        display_menu() # calls the display_menu() function
                        while True:
                            command = pyp.inputStr(prompt='Command: ') # gets the command that user chooses
                            command = command.lower()
                        if command == 'view':
                            view() # calls the view function
                        elif command == 'add':
                            add() # calls the add function
                        #elif command == 'del':
                            #delete()
                        elif command == 'exit':
                            print('Thank you for using the program. See you next time!')
                            break # breaks out of the loop
                        else:
                            print('Not a valid command. Please try again. \n')
                    except KeyError:
                        print('Key Error ')


                    def display_menu():
                        print('COMMAND MENU')
                        print('view - View username/email file')
                        print('add - Add username(s)/email(s) file')
                        #print('del - Delete a country')
                        print('exit - Exit program')
                        print()

                    def view():
                        myFile = open('directory.txt','r') # 'r' for read mode
                        myLine = myFile.readline() # reads one line of the file as a string
                        while myLine !='': # until there is nothing to read
                            print(myLine,end='') # end='' replaces the '\n' at the end
                            myLine = myFile.readline() # reads the next line
                        myFile.close()# closes the file

                    def add():
                        myFile = open('directory.txt','a') # 'a' for append mode
                        userInput = pyp.inputStr(prompt='''You have chosen to add username & emails to the dictionary.
                        Use this format...
                        username email@address.com,username email@address.com
                        (Separate each new contact's entry with a comma.) ''')
                        userInputList = userInput.split(',')# seperates the user input from comma
                        for i in range(len(userInputList)): # loop accross the list
                            myFile.write(userInputList[i].strip() + '\n') # strips extra spaces and write each of the username, email combination to the file appending new line at the end.
                        myFile.close() # closes the file


                    if __name__ == '__main__':
                        main()
